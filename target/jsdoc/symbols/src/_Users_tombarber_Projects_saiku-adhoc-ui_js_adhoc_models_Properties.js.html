1 /**
  2  * Maintains query properties like non empty and automatic execution
  3  */
  4 var Properties = Backbone.Model.extend({    
  5     initialize: function(args, options) {
  6         // Keep track of parent query
  7         this.query = options.query;
  8         
  9         // Update properties with defaults from settings
 10         this.properties = {};
 11         _.extend(this.properties, Settings.QUERY_PROPERTIES);
 12         this.update();
 13     },
 14     
 15     toggle: function(key) {
 16         // Toggle property
 17         this.properties[key] = this.properties[key] === 'true' ? 
 18             'false' : 'true';
 19         
 20         return this;
 21     },
 22     
 23     update: function() {
 24         // FIXME - this really sucks
 25         // Why can't we just use the body?
 26         this.attributes = {
 27             properties: _.template(
 28                     "<% _.each(properties, function(property, name) { %>" +
 29                     "<%= name %> <%= property %>\n" +
 30                     "<% }); %>"
 31                     )({ properties: this.properties })
 32         };
 33         this.save();
 34     },
 35     
 36     parse: function(response) {
 37         // FIXME - POST should return properties as well
 38         if (typeof response == "object") {
 39             _.extend(this.properties, response);
 40         }
 41 
 42         this.query.workspace.trigger('properties:loaded');
 43     },
 44     
 45     url: function() {
 46         return encodeURI(this.query.url() + "/properties");
 47     }
 48 });
