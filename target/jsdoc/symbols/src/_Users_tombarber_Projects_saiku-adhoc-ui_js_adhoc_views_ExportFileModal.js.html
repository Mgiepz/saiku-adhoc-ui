1 /**
  2  * Dialog for column configuration
  3  */
  4 var ExportFileModal = Modal.extend({
  5     type: "export",
  6     
  7     buttons: [
  8         { text: "Save", method: "save" },
  9         { text: "Cancel", method: "close" }
 10     ],
 11     
 12     events: {
 13         'click a': 'call' 
 14     },
 15     
 16     initialize: function(args) {
 17 
 18         // Initialize properties
 19         _.extend(this, args);
 20         this.options.title = "Export " + args.extension;
 21         this.message = "Fetching repository...";
 22         this.query = args.workspace.query;
 23         this.extension = args.extension;
 24         _.bindAll(this, "finished","post_render","save");
 25               
 26         // Resize when rendered
 27         this.bind('open', this.post_render);
 28 
 29 		this.render(); 
 30 
 31     },
 32 
 33     post_render: function(args) {
 34     	
 35        this.treeTemplate = _.template($("#template-filetree").html())(); 
 36 
 37     	$(this.el).find('.dialog_body').html(this.treeTemplate);
 38     	
 39     	$.ajaxSetup({traditional:true});
 40 
 41     	var url = "http://localhost:8080/pentaho/content/adhoc-service/ExploreFolder?fileExtensions=." + this.extension.toLowerCase();
 42     	
 43     	var that = this;
 44     	
 45     	 selectedFile = "";
 46           $('#container_id').fileTree(
 47           {
 48             root: '/',
 49             script: url,
 50             expandSpeed: 1000,
 51             collapseSpeed: 1000,
 52             multiFolder: false,
 53             folderClick:
 54             function(obj,folder){
 55               if($(".selectedFolder").length > 0)$(".selectedFolder").attr("class","");
 56               $(obj).attr("class","selectedFolder");
 57               that.selectedFolder = folder;//TODO:
 58             }
 59           },
 60           function(file) {
 61             that.selectedFile = file;
 62             $(".selectedFile").attr("class","");
 63             $("a[rel='" + file + "']").attr("class","selectedFile");
 64           });
 65 
 66         $(this.el).parents('.ui-dialog').css({ width: "200px" });
 67         
 68     },
 69     
 70     save: function() {
 71     	$.ajaxSetup({traditional:false});
 72     	
 73     	var file = $("#fileInput").val();
 74     	
 75     	if(file=='undefined' || this.selectedFolder == 'undefined'){
 76     		return false;
 77     	}
 78 	
 79 		var values = {"path": this.selectedFolder ,"file": file};
 80 
 81         this.query.action.post('/EXPORT/' + this.extension, { 
 82             success: this.finished,
 83             data: values
 84         });
 85 
 86         return false;
 87     },
 88     
 89     finished: function() {
 90     	$.ajaxSetup({traditional:false});
 91         $(this.el).dialog('destroy').remove();
 92     }
 93 });
