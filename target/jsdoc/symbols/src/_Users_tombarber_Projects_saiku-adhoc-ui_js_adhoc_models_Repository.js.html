1 /**
  2  * Repository query
  3  */
  4 var SavedQuery = Backbone.Model.extend({
  5     parse: function(response, XHR) {
  6         this.json = response.json;
  7     },
  8     
  9     url: function() {   
 10         return encodeURI(Settings.REST_URL + "/repository/query/" + this.get('name'));
 11     },
 12     
 13     move_query_to_workspace: function(model, response) {
 14         var query = new Query({ 
 15             json: model.json
 16         }, {
 17             name: model.get('name')
 18         });
 19         
 20         var tab = Application.tabs.add(new Workspace({ query: query }));
 21     }
 22 });
 23 
 24 /**
 25  * Repository adapter
 26  */
 27 var Repository = Backbone.Collection.extend({
 28     model: SavedQuery,
 29     
 30     initialize: function(args, options) {
 31         this.dialog = options.dialog;
 32     },
 33     
 34     parse: function(response) {
 35         this.dialog.populate(response);
 36     },
 37     
 38     url: function() {
 39         return encodeURI(Settings.REST_URL + "repository/query");
 40     }
 41 });
