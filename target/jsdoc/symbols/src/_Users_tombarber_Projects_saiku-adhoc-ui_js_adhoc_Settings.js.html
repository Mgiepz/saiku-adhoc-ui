 23 /**
 24  * Extend settings with query parameters
 25  */
 26 Settings.GET = function () {
 27     var qs = document.location.search;
 28     qs = qs.split("+").join(" ");
 29     var params = {},
 30         tokens,
 31         re = /[?&]?([^=]+)=([^&]*)/g;
 32 
 33     while (tokens = re.exec(qs)) {
 34         var value = decodeURIComponent(tokens[2]);
 35         if (! isNaN(value)) value = parseInt(value);
 36         if (value === "true") value = true;
 37         if (value === "false") value = false;
 38         params[decodeURIComponent(tokens[1]).toUpperCase()]
 39             = value;
 40     }
 41 
 42     return params;
 43 }();
 44 _.extend(Settings, Settings.GET);
