1 var Session = Backbone.Model.extend({
  2 
  3     initialize: function() {
  4 
  5         // Attach a custom event bus to this model
  6         _.extend(this, Backbone.Events);
  7         _.bindAll(this, "process_session","templates_callback", "prefetch_md_models");
  8         
  9         // Check expiration on localStorage
 10         if (! (localStorage.getItem('expiration') > (new Date()).getTime())) {
 11             localStorage.clear();
 12         }
 13         
 14         // Set expiration on localStorage to one day in the future
 15         var expires = (new Date()).getTime() + 
 16             Settings.LOCALSTORAGE_EXPIRATION;
 17         localStorage.setItem('expiration', expires);
 18         
 19         // Create session and fetch domains information
 20         // calls getDomains on DomainDiscoverResource
 21         // load additional templates
 22     	$.get('web/../js/templates/editor-templates.htm', this.templates_callback(this));
 23        
 24         return false;
 25     },
 26     
 27     templates_callback: function(session) {
 28     	return function(templates) {
 29 			$('body').append(templates);
 30 			session.fetch({ success: session.process_session });
 31 		};		
 32 	},
 33        
 34     error: function() {
 35         $(this.form.el).dialog('open');
 36     },
 37     
 38 
 39     process_session: function(mdModelInfo, response) {
 40   	
 41         // Save session in localStorage for other tabs to use
 42         if (localStorage && localStorage.getItem('session') === null) {
 43             localStorage.setItem('session', JSON.stringify(response));
 44         }
 45 
 46         // Generate model navigation for reuse
 47         this.model_navigation = _.template($("#template-models").html())({
 48             mdModelInfos: response
 49         });
 50 
 51         // Create domain objects
 52         this.mdModels = {};
 53         this.mdModelInfos = response;
 54         _.delay(this.prefetch_md_models, 200);
 55         
 56         // Show UI
 57         $(Application.toolbar.el).prependTo($("#header"));
 58         $("#header").show();
 59         Application.ui.unblock();
 60         
 61         // Add initial tab
 62         Application.tabs.render();
 63         Application.tabs.add(new Workspace());
 64         
 65         // Notify the rest of the application that login was successful
 66         Application.events.trigger('session:new', {
 67             session: this
 68         });
 69     },
 70     
 71     prefetch_md_models: function() {
 72     	
 73         if (! this.mdModels) {
 74             Log.log(JSON.stringify({
 75                 Message: "categories not initialized",
 76                 Session: JSON.stringify(this)
 77             }));

 85         	var key = mdModelInfo.domainId  + "/" + mdModelInfo.modelId;
 86         	var path = key;

 88         	
 89         	//var path = encodeURIComponent(mdModelInfo.domainId) + "/" + mdModelInfo.modelId;
 90 
 91 			if (localStorage && localStorage.getItem("md_model." + key) !== null){
 92 				this.mdModels[key] = new MdModel(JSON.parse(localStorage.getItem("md_model." + key)));
 93 			}else{
 94 				this.mdModels[key] = new MdModel({ path: path, key: key });
 95                 this.mdModels[key].fetch();	
 96 			}
 97         }	
 98         
 99         // Start routing
100         if (Backbone.history) {
101             Backbone.history.start();
102         }
103
