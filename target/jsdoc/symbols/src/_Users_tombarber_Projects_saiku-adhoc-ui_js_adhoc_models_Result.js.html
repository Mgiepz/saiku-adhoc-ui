1 /**
  2  * Holds the resultset for a query, and notifies plugins when resultset updated
  3  */
  4 var Result = Backbone.Model.extend({
  5     initialize: function(args, options) {
  6         // Keep reference to query
  7         this.query = options.query;
  8     },
  9     
 10     parse: function(response) {
 11 
 12         this.query.workspace.trigger('query:result', {
 13             workspace: this.query.workspace,
 14             data: response
 15         });
 16         
 17                 // Show the UI if hidden
 18         Application.ui.unblock();
 19       
 20     },
 21     
 22     url: function() {
 23     	return encodeURI(this.query.url() + "/result");
 24     }
 25 });
