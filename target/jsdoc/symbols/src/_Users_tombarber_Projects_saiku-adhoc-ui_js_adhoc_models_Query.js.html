1 /**
  2  * Workspace query
  3  */
  4 var Query = Backbone.Model.extend({
  5     initialize: function(args, options) {
  6     	
  7     	_.extend(this, options);
  8         
  9         // Bind `this`
 10         _.bindAll(this, "run", "move_dimension", "reflect_properties");
 11         
 12         // Generate a unique query id
 13         this.uuid = 'xxxxxxxx-xxxx-xxxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, 
 14             function (c) {
 15                 var r = Math.random() * 16 | 0,
 16                 v = c == 'x' ? r : (r & 0x3 | 0x8);
 17                 return v.toString(16);
 18             }).toUpperCase();
 19             
 20             
 21         this.reportPerspective = true;
 22         
 23         this.action = new QueryAction({}, { query: this });
 24         this.result = new Result({}, { query: this });
 25 
 26         this.reportresult = new ReportResult({}, { query: this });
 27         this.inplace = new InplaceEdit({}, { query: this });
 28 
 29     },
 30     
 31     parse: function(response) {
 32 
 33     	 // Fetch initial properties from server
 34         if (! this.properties) {
 35             this.properties = new Properties({}, { query: this });
 36         } else {
 37             this.properties.fetch({
 38                 success: this.reflect_properties
 39             });
 40         }
 41        
 42     },
 43     
 44     reflect_properties: function() {
 45         this.workspace.trigger('properties:loaded');
 46     },
 47 
 48     run: function(force) {
 49     	
 50         if (! this.properties.properties['saiku.adhoc.query.automatic_execution'] &&
 51             ! force) {
 52             return;
 53         }
 54 
 55 		Application.ui.block("Rendering Report");
 56 
 57 	//this has to be the other way round
 58 		if(this.reportPerspective){
 59 			this.reportresult.fetch();
 60 		}else{
 61 			this.result.fetch();
 62 		}
 63 
 64    },
 65     
 66     move_dimension: function(dimension, $target_el, index) {
 67         $(this.workspace.el).find('.run').removeClass('disabled_toolbar');
 68         
 69         var target = '';
 70         if ($target_el.hasClass('columns')) target = "COLUMNS";
 71         if ($target_el.hasClass('group')) target = "GROUP";
 72         if ($target_el.hasClass('filter')) target = "FILTER";
 73 
 74         var index = $target_el.find('li.ui-draggable').index(
 75                 $target_el.find('a[href="#' + dimension + '"]').parent() );
 76         
 77         var url = "/" + target + "/" + dimension + "/POSITION/" + index;
 78 
 79        // var $element = $target_el.find('li.ui-draggable');
 80         
 81         var uid = $target_el.find('a[href="#' + dimension + '"]').parent().attr('id');
 82         
 83         this.action.post(url, {
 84         	data: {
 85                 position: index,
 86                 uid: uid
 87             },
 88             success: function() {
 89             		 if (this.query.properties
 90                     .properties['saiku.adhoc.query.automatic_execution'] === 'true' && target != 'FILTER') {
 91                     this.query.run();
 92                 }
 93             }
 94         });
 95     },
 96     
 97     url: function() {
 98         return encodeURI(Settings.REST_URL + "/query/" + this.uuid);
 99     }
100 });
