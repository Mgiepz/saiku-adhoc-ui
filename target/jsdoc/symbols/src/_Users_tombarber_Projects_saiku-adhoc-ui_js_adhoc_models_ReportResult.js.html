1 /**
  2  * Holds the resultset for a query, and notifies plugins when resultset updated
  3  */
  4 var ReportResult = Backbone.Model.extend({
  5     initialize: function(args, options) {
  6         // Keep reference to query
  7         this.query = options.query;
  8        
  9     },
 10     
 11     parse: function(response) {
 12 
 13 	this.query.workspace.trigger('query:report', {
 14             workspace: this.query.workspace,
 15             data: response
 16         });
 17         
 18         // Show the UI if hidden
 19         Application.ui.unblock();
 20         
 21     },
 22     
 23     url: function() {
 24     	var template = this.query.template!=null ? this.query.template : "default";
 25     	var page = this.query.page!=null ? this.query.page : "1";
 26     	return encodeURI(this.query.url() + "/report/" + template + "/" + page);
 27     }
 28 });
