1 /**
  2  * Controls the appearance and behavior of the dimension list
  3  * 
  4  * This is where drag and drop lives
  5  */
  6 var MdModelList = Backbone.View.extend({
  7     events: {
  8         'click span': 'select',
  9         'click a': 'select'
 10     },
 11     
 12     initialize: function(args) {
 13         // Don't lose this
 14         _.bindAll(this, "render", "load_md_model");
 15         
 16         // Bind parent element
 17         this.workspace = args.workspace;
 18         //this.mdModel = args.mdModel.split('/')[1];
 19         this.mdModel = args.mdModel;
 20 
 21      	// Fetch from the server if we haven't already
 22         if (args.mdModel && args.mdModel.has('template')) {
 23             this.load_md_model();
 24         } else if (! args.mdModel){
 25         	alert("..." + args.mdModel);
 26             $(this.el).html('Could not load model. Please log out and log in again.');
 27         } else {
 28             $(this.el).html('Loading...');
 29             args.mdModel.fetch({ success: this.load_md_model });
 30         }
 31     },
 32     
 33     load_md_model: function() {
 34         this.template = this.mdModel.get('template');
 35         this.render(); 
 36         this.workspace.trigger('models:loaded');
 37     },
 38     
 39     render: function() {
 40         // Pull the HTML from cache and hide all dimensions
 41         $(this.el).html(this.template)
 42             .find('.hide').hide().removeClass('hide');
 43 
 44 		
 45 		var colprototype =	"<li class=\"hide\">" +
 46 			"<a title=\"calc_column\" class=\"dimension\" href=\"#CALCULATED/COLUMN/UID\">calc_column</a>" +
 47 			"</li>";
 48 
 49         $(this.el).append(colprototype);
 50            
 51              
 52         // Add draggable behavior
 53         $(this.el).find('.measure,.dimension').parent('li').draggable({
 54             cancel: '.not-draggable, .hierarchy',
 55             connectToSortable: $(this.workspace.el)
 56                 .find('.columns > ul, .group > ul, .filter > ul'),
 57             helper: 'clone',
 58             opacity: 0.60,
 59             tolerance: 'pointer',
 60             cursorAt: {
 61                 top: 10,
 62                 left: 35
 63             }
 64         });
 65     },
 66     
 67     select: function(event) {
 68         var $target = $(event.target).hasClass('root')
 69             ? $(event.target) : $(event.target).parent().find('span');
 70         if ($target.hasClass('root')) {
 71             $target.find('a').toggleClass('folder_collapsed').toggleClass('folder_expand');
 72             $target.toggleClass('collapsed').toggleClass('expand');
 73             $target.parents('li').find('ul').children('li').toggle();
 74         }
 75         
 76         return false;
 77     }
 78 });
