 11 var WorkspaceToolbar = Backbone.View.extend({
 12     enabled: false,
 13     events: {
 14         'click a': 'call',
 15         'change select' : 'changed'    
 16     },
 17     
 18     
 19     
 20     initialize: function(args) {
 21         // Keep track of parent workspace
 22         this.workspace = args.workspace;
 23         
 24         // Maintain `this` in callbacks 
 25          _.bindAll(this, "call", "changed", "reflect_properties", "run_query", "toggle_report", "calculated_column");
 26         
 27         // Redraw the toolbar to reflect properties
 28         this.workspace.bind('properties:loaded', this.reflect_properties);
 29         
 30         // Fire off workspace event
 31         this.workspace.trigger('workspace:toolbar:render', { 
 32             workspace: this.workspace
 33         });
 34         
 35         // Activate buttons when a new query is created or run
 36         this.workspace.bind('query:new', this.activate_buttons);
 37         this.workspace.bind('query:result', this.activate_buttons);
 38     },
 39     
 40     
 41     activate_buttons: function(args) {
 42         if (args.data && args.data.length > 0) {
 43             $(args.workspace.toolbar.el).find('.button')
 44                 .removeClass('disabled_toolbar');            
 45         } else {
 46             $(args.workspace.toolbar.el).find('.button')
 47                 .addClass('disabled_toolbar');
 48             $(args.workspace.toolbar.el)
 49                 .find('.auto,.formula,.toggle_fields,.toggle_sidebar, .export_xls, .export_pdf, .export_csv,.cda,.prpt, .report')
 50                 .removeClass('disabled_toolbar');
 51         }
 52     },
 53     
 54     template: function() {
 55        return _.template($("#template-workspace-toolbar").html())();
 56     },
 57     
 58     render: function() {
 59         $(this.el).html(this.template());
 60 
 61         return this; 
 62     },
 63     
 64 
 65 	changed: function(event){
 66 		
 67 		var that = this;
 68 		
 69 		this.workspace.query.action.get("/ROWLIMIT/" + $(event.target).val() , { 
 70             success: function(){
 71             	that.workspace.query.run(true);
 72             }
 73         });
 74 	},
 75 
 76     call: function(event) {
 77         //Determine callback
 78         var callback = event.target.hash.replace('#', '');
 79         
 80         //Attempt to call callback
 81         if (! $(event.target).hasClass('disabled_toolbar') && this[callback]) {
 82            this[callback](event);
 83         }
 84         
 85         return false;
 86     },
 87     
 88     run_query: function(event) {
 89         this.workspace.query.run(true);
 90     },
 91 
 92     toggle_report: function(event) {
 93         this.workspace.toggle_report();
 94     },
 95 
 96     
 97     reflect_properties: function() {
 98         var properties = this.workspace.query.properties ?
 99             this.workspace.query.properties.properties : Settings.QUERY_PROPERTIES;
100             
101         // Set properties appropriately
102         if (properties['saiku.adhoc.query.automatic_execution'] === 'true') {
103             $(this.el).find('.auto').addClass('on');
104         }
105     },
106 
107     save_query: function(event) {
108         if (this.workspace.query) {
109             (new SaveQuery({ query: this.workspace.query })).render().open();
110         }
111     },
112 
113     automatic_execution: function(event) {
114         // Change property
115         this.workspace.query.properties
116             .toggle('saiku.adhoc.query.automatic_execution').update();
117         
118         // Toggle state of button
119         $(event.target).toggleClass('on');
120     },
121     
122     toggle_fields: function(event) {
123         $(this.workspace.el).find('.workspace_fields').toggle();
124     },
125     
126     toggle_sidebar: function() {
127         this.workspace.toggle_sidebar();
128     },
129     
130    setup_report: function(event) {
131    	    (new ReportSetupModal({
132             workspace: this.workspace
133         })).open();     
134     },
135    
136    add_union: function(event) {
137         alert("Union Queries are not yet implemented, sorry!");
138     },
139    
140    add_join: function(event) {
141         alert("Joined Queries are not yet implemented, sorry!");
142     },
143 
144    calculated_column: function(event) {
145    	
146    	     // Launch column config dialog
147         (new ColumnConfigModal({
148             //target: $target,
149             index: -1,
150             name: 'calculated',
151             key: 'CATEGORY/CALCULATED/COLUMN/NEW',
152             workspace: this.workspace
153         })).open();
154 
155    		//this.workspace.query.add_calculated_column();
156         //alert("Calculated Columns are not yet implemented, sorry!");
157     },    
158  
159     export_xls: function(event) {
160         window.location = Settings.REST_URL +
161             "/export/" + this.workspace.query.uuid + "/xls";
162     },
163     
164     export_csv: function(event) {
165         window.location = Settings.REST_URL +
166             "/export/" + this.workspace.query.uuid + "/csv";
167     },
168 
169     export_pdf: function(event) {
170         window.location.href = Settings.REST_URL +
171             "/export/" + this.workspace.query.uuid + "/pdf";
172     },
173     
174     export_cda: function(event) {
175         (new ExportFileModal({
176             workspace: this.workspace,
177             extension: "CDA"
178         })).open();   
179     },    
180 
181     export_prpt: function(event) {
182         (new ExportFileModal({
183             workspace: this.workspace,
184             extension: "PRPT"
185         })).open();   
186     } 
187 
188 
189 });
