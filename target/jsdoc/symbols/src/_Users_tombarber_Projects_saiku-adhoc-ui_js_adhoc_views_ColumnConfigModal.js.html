1 /**
  2  * Dialog for column configuration
  3  */
  4 var ColumnConfigModal = Modal.extend({
  5     type: "columnconfig",
  6     
  7     buttons: [
  8         { text: "Save", method: "save" },
  9         { text: "Cancel", method: "close" }
 10     ],
 11     
 12     events: {
 13         'click a': 'call' 
 14     },
 15     
 16 	changed: function(evt) {
 17 		var target = $(evt.currentTarget);
 18   
 19     },    
 20     
 21     initialize: function(args) {
 22         // Initialize properties
 23         _.extend(this, args);
 24         this.options.title = "Setup Column " + this.name;
 25         this.message = "Fetching config...";
 26         this.show_unique = false;
 27         this.query = args.workspace.query;
 28         _.bindAll(this, "fetch_values", "populate", "finished", "changed", "add_calculated_column");
 29           
 30         // Resize when rendered
 31         this.bind('open', this.post_render);
 32         this.render();
 33         
 34       	this.category = args.key.split('/')[1];
 35         this.column = args.key.split('/')[3];
 36         this.index = args.index;
 37    
 38       if(!(this.category == 'CALCULATED' && this.column == 'NEW')){
 39       	this.fetch_values();
 40       }else{
 41       	
 42       	var temp = defaultCalcColumn;
 43       	
 44       	this.json = $.extend(true, {}, defaultCalcColumn);
 45 
 46       	this.populate();
 47       }
 48 
 49     },
 50 
 51     fetch_values: function(category, column) {
 52 
 53         this.workspace.query.action.get("/COLUMNS/CATEGORY/" + this.category + "/COLUMN/" + this.column + "/POSITION/" + this.index + "/config", { 
 54             success: this.populate
 55         });
 56     },
 57 
 58     populate: function(model, response) {
 59 
 60      	var template = _.template($("#template-column-setup").html())(this);
 61 
 62    	
 63      	if(response != null){
 64      		this.json = response;
 65      	}
 66      	
 67      	$(this.el).find('.dialog_body').html(template);
 68      	
 69      	$(this.el).find('#description').html(this.json.description);
 70      	
 71      	//formula element needs to be made visible
 72      	if(this.category == 'CALCULATED'){
 73      		$(this.el).find('#formula').removeClass('hide').find('.formula').val(this.json.formula);
 74      	}
 75 
 76      	$(this.el).find('#displayname input').val(this.json.name);
 77      	
 78      	$(this.el).find('#format input').val(this.json.formatMask);   	
 79      	if(this.json.fieldType=='Numeric'||this.json.fieldType=='Date'){
 80      		$(this.el).find('#format input').removeAttr('disabled');
 81      	}
 82      	
 83      	if(this.json.aggTypes!=null){
 84             for (var j = 0; j < this.json.aggTypes.length; j++) {
 85                 var value = this.json.aggTypes[j];
 86                     $("<option />").text(AggTypes[value])
 87                         .val(value)
 88                         .appendTo($(this.el).find('#aggregation select'));
 89            	}
 90  		} 
 91 
 92 		$(this.el).find('#aggregation select').val(this.json.selectedAggType);
 93 
 94         for (var value in AggTypes) {
 95               $("<option />").text(AggTypes[value]).val(value)
 96                 .appendTo($(this.el).find('#summary select'));    			
 97 		}
 98             	
 99      	$(this.el).find('#summary select').val(this.json.selectedSummaryType);
100 
101      	// Show dialog
102         Application.ui.unblock();
103      		
104     },
105     
106     post_render: function(args) {
107         $(args.modal.el).parents('.ui-dialog').css({ width: "200px" });
108     },
109     
110     save: function() {
111     	
112     	this.json.name = $(this.el).find('#displayname input').val();
113     	this.json.formatMask = $(this.el).find('#format input').val();   
114 
115 		if(!$(this.el).find('#formula').hasClass('hide')){
116 			this.json.formula = $(this.el).find('#formula .formula').val();   		
117 		};
118 
119     	this.json.selectedAggType = $(this.el).find('#aggregation select').val();   
120     	this.json.selectedSummaryType = $(this.el).find('#summary select').val();  
121 
122 
123     	this.json.hideRepeating = $(this.el).find('#show_unique').is(':checked');  
124     	
125     	//this.json.elementFormat.horizontalAlignment = $(this.el).find('#alignment select').val();   
126     	
127  		if(this.json.uid == null) this.json.uid = this.workspace.uniqueId('rpt-dtl-');
128     	
129         // Notify user that updates are in progress
130         var $loading = $("<div>Saving...</div>");
131         $(this.el).find('.dialog_body').children().hide();
132         $(this.el).find('.dialog_body').prepend($loading);
133        
134         // Notify server
135             this.query.action.post("/COLUMNS/CATEGORY/" + this.category + "/COLUMN/" + this.column + "/POSITION/" + this.index + "/config", { 
136             success: this.finished,
137             data: this.json// JSON.stringify(values)
138         });
139         
140         return false;
141     },
142     
143     add_calculated_column: function(){
144     	
145     	var $selections = $(this.workspace.el).find('.columns ul');
146 	 	
147 		var $logicalColumn = $('.category_tree')
148             .find('a[title="calc_column"]')
149             .parent();
150 
151         var $clone = $logicalColumn.clone()
152             .addClass('d_measure')
153             .addClass('.calculated')
154             .attr("id",this.json.uid)
155             .removeClass('hide');
156             
157             $clone.find('a[title="calc_column"]').attr("title",this.json.name).html(this.json.name)
158             .attr("href","abc");
159  
160             $clone.appendTo($selections);		 	
161    },
162     
163     finished: function() {
164     	
165       	if(this.category == 'CALCULATED' && this.column == 'NEW'){
166       		this.add_calculated_column();
167         }
168         
169         $(this.el).dialog('destroy').remove();
170         this.query.run();
171     }
172 });
