129     reflect_formatting: function(model, response) {
130     	
131     
132     	this.json = response;
133     	var format = this.json.format;
134 
135 		//reset elements
136     	$(this.el).removeClass('disabled_editor');
137     	$(this.el).find('select').removeAttr('disabled');

141

143 		$(this.el).find('.horz.align-' + horzAlignment).addClass('on');
144 
145 		var vertAlignment = format.verticalAlignment.toLowerCase();
146 		$(this.el).find('.vert.align-' + vertAlignment).addClass('on');
147 
148 		$(this.el).find('.sizeSelector select').val(format.fontSize);
149

161 		};

187     fetch_values: function(element, type) {
188  	
189  		//fetch from different url depending on type
190  		this.element = element;
191  	
192         this.workspace.query.action.get("/FORMAT/ELEMENT/" + element , { 
193             success: this.reflect_formatting
194         });
195     },
196 
197     save: function(model) {      
198         	// Notify server
199         this.workspace.query.action.post("/FORMAT/ELEMENT/" + this.element, { 
200             success: this.finished,
201             data: model
202         });
203         
204         return false;
205     },
206     
207     finished: function() {
208         this.query.run();
209     },
210 
211     call: function(event) {
212         //Determine callback
213         var callback = event.target.hash.replace('#', '');
214 
215         //Attempt to call callback
216         if (! $(event.target).hasClass('disabled_toolbar') && this[callback]) {
217            this[callback](event);
218         }
219         
220         return false;
221     },
