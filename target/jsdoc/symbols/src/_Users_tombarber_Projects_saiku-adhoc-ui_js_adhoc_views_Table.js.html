1 /**
  2  * Class which handles table rendering of resultsets
  3  */
  4 var Table = Backbone.View.extend({
  5     //tagName: "table",
  6     initialize: function(args) {
  7         this.workspace = args.workspace;
  8         
  9         // Bind table rendering to query result event
 10         _.bindAll(this, "render", "process_data");
 11         this.workspace.bind('query:result', this.render);
 12     },
 13     
 14     render: function(json) {
 15       
 16         _.delay(this.process_data, 0, json);
 17     },
 18     
 19     process_data: function(json) {
 20  
 21 	  var data = json.data;
 22 	  
 23 		// Check if there is data
 24     	if (data == null) {
 25            return this.no_results(json);
 26      	}
 27     
 28       var tableContents = data.resultset;
 29       var columnNames = [];
 30       
 31      
 32       for (column in data.metadata) {
 33         columnNames.push({"sTitle": data.metadata[column].colName});
 34       }
 35       $(this.el).empty();
 36       $(this.el).html( '<table cellpadding="0" cellspacing="0" border="0" class="display" id="contents"></table>' );
 37 
 38 					
 39       $('#contents').dataTable({
 40       "aaData": tableContents, 
 41       "aoColumns": columnNames,
 42       "bAutoWidth": true,
 43       "bLengthChange": false,
 44       "iDisplayLength": 30,
 45       "bFilter": false,
 46 	  "bSort": false,
 47       "bJQueryUI": false,
 48       "sPaginationType": "full_numbers"
 49       });
 50      
 51     },
 52     
 53     no_results: function(args) {
 54         $(args.workspace.el).find('.workspace_results table')
 55             .html('<tr><td>No results</td></tr>');
 56     },
 57     
 58     error: function(args) {
 59         $(args.workspace.el).find('.workspace_results table')
 60             .html('<tr><td>' + args.data[0][0].value + '</td></tr>');
 61     }
 62 });
