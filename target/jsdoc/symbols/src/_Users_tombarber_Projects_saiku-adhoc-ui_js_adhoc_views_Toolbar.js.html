1 /**
  2  * The global toolbar
  3  * 
  4  * With Buttons
  5  * - Append Join
  6  * - Append Union
  7  * - Create Output
  8  * - Reset
  9  * - Vielleicht auch für den Output standardmößig einen letzten Tab einfügen
 10  * - Oder Pro output einen Tab mit dem Entsprechenden Symbol
 11  */
 12 var Toolbar = Backbone.View.extend({
 13     tagName: "div",
 14     
 15     events: {
 16         'click #add_join': 'add_join',
 17         'click #add_union': 'add_union',
 18         'click #create_prpt_output': 'create_prpt_output',
 19     	'click #create_table_output': 'create_table_output',
 20     	'click #create_xls_output': 'create_xls_output',
 21     },
 22     
 23     template: function() {
 24         return _.template("<ul>" + 
 25                     "<li><a id='new_query' href='#new_query' title='New query' class='new_tab i18n sprite'></a></li>" +
 26                     "<li class='separator'> </li>" +
 27                     "<li><a id='open_query' href='#open_query' title='Open query' class='open_query i18n sprite'></a></li>" + 
 28                     "<li class='separator'> </li>" +
 29                     "<li><a id='logout' href='#logout' title='Logout' class='logout i18n sprite'></a></li>" +
 30                     "<li><a id='about' href='#about' title='About' class='about i18n sprite'></a></li>" +
 31                     "<li class='separator'> </li>" +
 32                     "<li><a id='issue_tracker' href='#issue_tracker' title='Issue Tracker' class='bug i18n sprite'></a></li>" +
 33                 "</ul>" +
 34                 "<h1 id='logo'><a href='http://www.analytical-labs.com/' title='Saiku - Next Generation Open Source Analytics' class='sprite'>Saiku</a></h1>"
 35             )(this);
 36     },
 37     
 38     initialize: function() {
 39         this.render();
 40     },
 41     
 42     render: function() {
 43         $(this.el).attr('id', 'toolbar')
 44             .html(this.template());
 45         
 46         // Trigger render event on toolbar so plugins can register buttons
 47         Application.events.trigger('toolbar:render', { toolbar: this });
 48         
 49         return this;
 50     },
 51     
 52     /**
 53      * Add a new tab to the interface
 54      */
 55     new_query: function() {
 56         Saiku.tabs.add(new Workspace());
 57         return false;
 58     },
 59     
 60     /**
 61      * Open a query from the repository into a new tab
 62      */
 63     open_query: function() {
 64         var dialog = _.detect(Saiku.tabs._tabs, function(tab) {
 65             return tab.content instanceof OpenQuery;
 66         });
 67         
 68         if (dialog) {
 69             dialog.select();
 70         } else {
 71             Saiku.tabs.add(new OpenQuery());
 72         }
 73         
 74         return false;
 75     },
 76     
 77     /**
 78      * Clear the current session and show the login window
 79      */
 80     logout: function() {
 81         Saiku.session.logout();
 82     },
 83     
 84     /**
 85      * Show the credits dialog
 86      */
 87     about: function() {
 88         (new AboutModal).render().open();
 89         return false;
 90     },
 91     
 92     /**
 93      * Go to the issue tracker
 94      */
 95     issue_tracker: function() {
 96         window.open('http://projects.analytical-labs.com/projects/saiku/issues/new');
 97         return false;
 98     }
 99 });
