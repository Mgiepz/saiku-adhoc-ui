 65     adjust: function() {

 67         $separator = $(this.el).find('.sidebar_separator');
 68         $separator.height($("body").height() - 87);
 69         $(this.el).find('.sidebar').height($("body").height() - 87);
 70         
 71         // Adjust the dimensions of the results window

 73             width: $(document).width() - $(this.el).find('.sidebar').width() - 30,
 74             height: $(document).height() - $("#header").height() -
 75                 $(this.el).find('.workspace_toolbar').height() - 
 76                 $(this.el).find('.workspace_fields').height() - 40
 77         });

204     new_query: function() {
205         // Delete the existing query
206 
207         if (this.query) {
208             this.query.destroy();
209         }
210         
211         
212         // Initialize the new query
213         this.selected_model = $(this.el).find('.mdModels').val();
214 
215         var mModelInfo = this.selected_model;
216         
217         var parsedModelInfo = this.selected_model.split('/');
218         
219 
220         this.query = new Query({
221         	domainId: parsedModelInfo[0], 
222             modelId: parsedModelInfo[1]
223         }, {
224             workspace: this
225         });
226      
227         // Save the query to the server and init the UI
228         this.query.save();
229           
230         this.init_query();

233     init_query: function() {

236 			var json = JSON.parse(this.query.get('json'));
237 			this.selected_model = json.clientModelSelection;
238 		}

242             this.selected_model = this.query.get('model');

244         }

247                 .val(this.selected_model);

253         	
254         //alert("looking " + this.selected_model);

256         // Create new DimensionList and MeasureList

258             workspace: this,
