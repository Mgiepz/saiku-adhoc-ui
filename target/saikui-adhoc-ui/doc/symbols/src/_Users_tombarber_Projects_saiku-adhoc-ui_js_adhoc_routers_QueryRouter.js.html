1 /**
  2  * Router for opening query when session is initialized
  3  */
  4 var QueryRouter = Backbone.Router.extend({
  5     routes: {
  6         'query/open/:query_name': 'open_query'
  7     },
  8     
  9     open_query: function(query_name) {
 10         Settings.ACTION = "OPEN_QUERY";
 11         var options = { 
 12             name: query_name,
 13             solution: Settings.GET.SOLUTION || "",
 14             path: Settings.GET.PATH || "",
 15             action: Settings.GET.ACTION || "",
 16             biplugin: true
 17         };
 18         var query = new SavedQuery(options);
 19         
 20         var args = "name=" + options.name
 21         + "&solution=" + options.solution
 22         + "&path=" + options.path
 23         + "&action=" + options.action
 24         
 25         query.fetch({data: args, success: query.move_query_to_workspace });
 26     }
 27 });
 28 
 29 Application.routers.push(new QueryRouter());
