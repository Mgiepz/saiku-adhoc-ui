1 /**
  2  * The base class for all modal dialogs
  3  */
  4 var Modal = Backbone.View.extend({
  5     tagName: "div",
  6     className: "dialog",
  7     type: "modal",
  8     message: "Put content here",
  9     
 10     options: {
 11         autoOpen: false,
 12         modal: true,
 13         title: "Modal dialog",
 14         resizable: false,
 15         draggable: false
 16     },
 17     
 18     events: {
 19         'click a': 'call'
 20     },
 21     
 22     buttons: [
 23         { text: "OK", method: "close" }
 24     ],
 25     
 26     template: function() {
 27         return _.template("<div class='dialog_icon'></div>" +
 28                 "<div class='dialog_body'><%= message %></div>" +
 29         		"<div class='dialog_footer'>" +
 30             "<% _.each(buttons, function(button) { %>" +
 31                 "<a class='form_button' href='#<%= button.method %>'> <%= button.text %> </a>" +
 32             "<% }); %>" +
 33             "</div>")(this);
 34     },
 35     
 36     initialize: function(args) {
 37         _.extend(this, args);
 38         _.bindAll(this, "call");
 39         _.extend(this, Backbone.Events);
 40     },
 41     
 42     render: function() {
 43         $(this.el).html(this.template())
 44             .addClass("dialog_" + this.type)
 45             .dialog(this.options);
 46         return this;
 47     },
 48     
 49     call: function(event) {
 50         // Determine callback
 51         var callback = event.target.hash.replace('#', '');
 52         
 53         // Attempt to call callback
 54         if (! $(event.target).hasClass('disabled_toolbar') && this[callback]) {
 55             this[callback](event);
 56         }
 57         
 58         return false;
 59     },
 60     
 61     open: function() {
 62         $(this.el).dialog('open');
 63         this.trigger('open', { modal: this });
 64         return this;
 65     },
 66     
 67     close: function() {
 68         $(this.el).dialog('destroy').remove();
 69         return false;
 70     }
 71 });
