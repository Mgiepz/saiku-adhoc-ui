1 /**
  2  * Model which handles "special" actions against the query
  3  * Ex.: selections, swap axis, mdx
  4  */
  5 var QueryAction = Backbone.Model.extend({
  6     initialize: function(args, options) {
  7         // Keep track of query
  8         this.query = options.query;
  9         
 10         // Set default url
 11         this.url = this.query.url;
 12     },
 13     
 14     get: function(action, options) {
 15         this.handle("fetch", action, options);
 16     },
 17     
 18     post: function(action, options) {
 19         this.handle("save", action, options);
 20     },
 21     
 22     put: function(action, options) {
 23         this.id = _.uniqueId('queryaction_');
 24         this.handle("save", action, options);
 25         delete this.id;
 26     },
 27     
 28     del: function(action, options) {
 29         this.id = _.uniqueId('queryaction_');
 30         this.handle("delete", action, options);
 31         delete this.id;
 32     },
 33     
 34     // Call arbitrary actions against the query
 35     handle: function(method, action, options) {
 36         // Set query action
 37         this.url = this.query.url() + escape(action);
 38         
 39         // Clear out old attributes
 40         this.attributes = options.data? options.data : {};
 41         
 42         // Initiate action
 43         if (method == "save") {
 44             // Handle response from server
 45             this.parse = options.success;
 46             
 47             this.save();
 48         } else if (method == "delete") {
 49             this.destroy(options);
 50         } else if (method == "fetch") {
 51             this.parse = function() {};
 52             this.fetch(options);
 53         }
 54     }
 55 });
