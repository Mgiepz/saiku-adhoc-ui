1 /**
  2  * Class which handles individual tabs
  3  */
  4 var Tab = Backbone.View.extend({
  5     tagName: 'li',
  6     
  7     events: {
  8         'click a': 'select',
  9         'mousedown a': 'remove',
 10         'click .close_tab': 'remove'
 11     },
 12     
 13     template: function() {        
 14         // Create tab
 15         return _.template("<a href='#<%= id %>'><%= caption %></a>" +
 16                 "<span class='close_tab sprite'>Close tab</span>")
 17             ({
 18                 id: this.id,
 19                 caption: this.caption
 20             });
 21     },
 22     
 23     /**
 24      * Assign a unique ID and assign a Backbone view 
 25      * to handle the tab's contents
 26      * @param args
 27      */
 28     initialize: function(args) {
 29         _.extend(this, Backbone.Events);
 30         _.extend(this, args);
 31         this.content.tab = this;
 32         this.caption = this.content.caption();
 33         this.id = _.uniqueId('tab_');
 34     },
 35     
 36     /**
 37      * Render the tab and its contents
 38      * @returns tab
 39      */
 40     render: function() {
 41         // Render the content
 42         this.content.render();
 43         
 44         // Generate the element
 45         $(this.el).html(this.template());
 46         
 47         return this;
 48     },
 49     
 50     /**
 51      * Destroy any data associated with this tab and ensure proper
 52      * garbage collection to avoid memory leaks
 53      */
 54     destroy: function() {
 55         // Delete data
 56         if (this.content && this.content.query) {
 57             this.content.query.destroy();
 58         }
 59     },
 60     
 61     /**
 62      * Select a tab
 63      * @param el
 64      */
 65     select: function() {
 66         // Deselect all tabs
 67         this.parent.select(this);
 68         
 69         // Select the selected tab
 70         $(this.el).addClass('selected');
 71         
 72         // Trigger select event
 73         this.trigger('tab:select');
 74         
 75         return false;
 76     },
 77     
 78     /**
 79      * Remove a tab
 80      * @returns {Boolean}
 81      */
 82     remove: function(event) {
 83         if (!event || event.which === 2 || $(event.target).hasClass('close_tab')) {
 84             // Remove the tab element
 85             $(this.el).remove();
 86             
 87             // Remove the tab
 88             try {
 89                 this.destroy();
 90             } catch (e) {
 91                 Log.log(JSON.stringify({
 92                     Message: "Tab could not be removed",
 93                     Tab: JSON.stringify(this)
 94                 }));
 95             }
 96             
 97             // Remote the tab object from the container
 98             this.parent.remove(this);
 99         }
100         
101         return false;
102     }
103 });

105 /**
106  * Class which controls tab pager
107  */
108 var TabPager = Backbone.View.extend({
109     className: 'pager_contents',
110     events: {
111         'click a': 'select'
112     },
113     
114     initialize: function(args) {
115         this.tabset = args.tabset;
116         $(this.el).hide().appendTo('body');
117     },
118     
119     render: function() {
120         var pager = "";
121         for (var i = 0; i < this.tabset._tabs.length; i++) {
122             pager += "<a href='#" + i + "'>" + 
123                 this.tabset._tabs[i].caption + "</a><br />";
124         }
125         $(this.el).html(pager);
126     },
127     
128     select: function(event) {
129         var index = $(event.target).attr('href').replace('#', '');
130         this.tabset._tabs[index].select();
131         $(this.el).hide();
132         event.preventDefault();
133         return false;
134     }
135 });

137 /**
138  * Class which controls the tab collection
139  */
140 var TabSet = Backbone.View.extend({
141     className: 'tabs',
142     queryCount: 0,
143     
144     events: { 'click a.pager': 'togglePager' },
145     
146     _tabs: [],
147     
148     /**
149      * Render the tab containers
150      * @returns tab_container
151      */
152     render: function() {
153         $(this.el).html('<a href="#pager" class="pager sprite"></a><ul></ul>')
154             .appendTo($('#header'));
155         this.content = $('<div id="tab_panel">').appendTo($('body'));
156         this.pager = new TabPager({ tabset: this });
157         return this;
158     },
159     
160     /**
161      * Add a tab to the collection
162      * @param tab
163      */
164     add: function(content) {
165         // Add it to the set
166         var tab = new Tab({ content: content });
167         this._tabs.push(tab);
168         this.queryCount++;
169         tab.parent = this;
170         
171         // Render it in the background, then select it
172         tab.render().select();
173         $(tab.el).appendTo($(this.el).find('ul'));
174         
175         // Trigger add event on session
176         Application.session.trigger('tab:add', { tab: tab });
177         this.pager.render();
178         
179         return tab;
180     },
181     
182     /**
183      * Select a tab, and move its contents to the tab panel
184      * @param tab
185      */
186     select: function(tab) {
187         // Clear selections
188         $(this.el).find('li').removeClass('selected');
189         
190         // Replace the contents of the tab panel with the new content
191         this.content.children().detach();
192         this.content.append($(tab.content.el));
193     },
194     
195     /**
196      * Remove a tab from the collection
197      * @param tab
198      */
199     remove: function(tab) {
200         // Add another tab if the last one has been deleted
201         if (this._tabs.length == 1) {
202             return;
203         }
204         
205         for (var i = 0; i < this._tabs.length; i++) {
206             if (this._tabs[i] == tab) {
207                 // Remove the element
208                 this._tabs.splice(i, 1);
209                 Application.session.trigger('tab:remove', { tab: tab });
210                 this.pager.render();
211                 
212                 // Select the previous, or first tab
213                 var next = this._tabs[i] ? i : 0;
214                 this._tabs[next].select();
215             }
216         }
217     },
218     
219     togglePager: function() {
220         $(this.pager.el).toggle();
221         return false;
222     }
223 });
