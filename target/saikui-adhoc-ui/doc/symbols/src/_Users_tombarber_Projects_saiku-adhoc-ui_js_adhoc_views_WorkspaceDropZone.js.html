1 /**
  2  * 
  3  * Sets up workspace drop zones for DnD and other interaction
  4  */
  5 var WorkspaceDropZone = Backbone.View.extend({
  6     template: function() {
  7         return _.template($("#template-workspace-dropzones").html())();
  8     },
  9     
 10     events: {
 11         'sortstop': 'select_dimension',
 12         'click a': 'selections',
 13         'click span.sort': 'sort'
 14     },
 15     
 16     initialize: function(args) {
 17         // Keep track of parent workspace
 18         this.workspace = args.workspace;
 19         
 20         // Maintain `this` in jQuery event handlers
 21         _.bindAll(this, "select_dimension", "move_dimension", 
 22                 "remove_dimension", "sort");
 23     },
 24     
 25     render: function() {
 26         // Generate drop zones from template
 27      
 28         $(this.el).html(this.template());
 29         
 30         // Activate drop zones
 31         $(this.el).find('.connectable').sortable({
 32             connectWith: $(this.el).find('.connectable'),
 33             cursorAt: {
 34                 top: 10,
 35                 left: 35
 36             },
 37             forcePlaceholderSize: true,
 38             items: '> li',
 39             opacity: 0.60,
 40             placeholder: 'placeholder',
 41             tolerance: 'pointer'
 42         });
 43         
 44         return this; 
 45     },
 46     
 47     select_dimension: function(event, ui) {
 48         // Short circuit if this is a move
 49         if (ui.item.hasClass('d_measure') ||
 50                 ui.item.hasClass('d_dimension')) {
 51             this.move_dimension(event, ui);
 52             return;
 53         }
 54         
 55         // Make the element and its parent bold
 56         var original_href = ui.item.find('a').attr('href');
 57         var $original = $(this.workspace.el).find('.sidebar')
 58             .find('a[href="' + original_href + '"]').parent('li');
 59         $original
 60             .css({fontWeight: "bold"})
 61             //.draggable('disable')
 62             ;
 63         $original.parents('.parent_dimension')
 64             .find('.root')
 65             .css({fontWeight: "bold"});
 66         
 67         // Wrap with the appropriate parent element
 68         if (ui.item.find('a').hasClass('dimension')) {
 69             
 70             //var $icon = $("<span />").addClass('sprite');
 71             if(ui.item.parents('.fields_list').attr('title')=='COLUMNS' || 
 72             	ui.item.parents('.fields_list').attr('title')=='GROUPS'){
 73             	var $icon = $("<span />").addClass('sort').addClass('none');
 74             }
 75             else{
 76             	$icon = $("<span />").addClass('sprite');
 77             }
 78             
 79             ui.item.addClass('d_dimension').prepend($icon);
 80         } else {
 81             ui.item.addClass('d_measure');
 82         }
 83 
 84         // Notify the model of the change
 85         var dimension = ui.item.find('a').attr('href').replace('#', '');
 86         var index = ui.item.parent('.connectable').children().index(ui.item);
 87         
 88         //ui.item.attr('id', _.uniqueId('col'));
 89         
 90         //uid- independent of the dropzone
 91         ui.item.attr('id', this.workspace.uniqueId('uid-'));
 92         
 93         /*
 94         if(ui.item.parents('.fields_list').attr('title')=='COLUMNS'){
 95         	ui.item.attr('id', this.workspace.uniqueId('rpt-dtl-'));
 96         }else if(ui.item.parents('.fields_list').attr('title')=='GROUPS'){
 97         	ui.item.attr('id', this.workspace.uniqueId('rpt-grp-'));
 98         }
 99         */
100         
101 
102         this.workspace.query.move_dimension(dimension, 
103                 $(event.target).parent(), index);
104 
105       // Prevent workspace from getting this event
106         event.stopPropagation();
107     },
108     
109     move_dimension: function(event, ui) {        
110         // Notify the model of the change
111         var dimension = ui.item.find('a').attr('href').replace('#', '');
112         var index = ui.item.parent('.connectable').children().index(ui.item);
113         if (! ui.item.hasClass('deleted')) {
114             this.workspace.query.move_dimension(dimension, 
115                 //$(event.target).parent()
116                 ui.item.parents('.fields_list_body')
117                 , index);
118         }
119   
120         // Prevent workspace from getting this event
121         event.stopPropagation();
122         return false;
123     },
124     
125     remove_dimension: function(event, ui) {
126         // Reenable original element
127         var original_href = ui.draggable.find('a').attr('href');
128         var $original = $(this.workspace.el).find('.sidebar')
129             .find('a[href="' + original_href + '"]').parent('li');
130         $original
131             .draggable('enable')
132             .css({ fontWeight: 'normal' });
133         
134         // Unhighlight the parent if applicable
135         if ($original.parents('.parent_dimension')
136                 .children().children('.ui-state-disabled').length === 0) {
137             $original.parents('.parent_dimension')
138                 .find('.root')
139                 .css({fontWeight: "normal"});
140         }
141         
142         // Notify server
143         var target = '';
144         var dimension = original_href.replace('#', '');
145         $target_el = ui.draggable.parent().parent('div.fields_list_body');
146 
147         if ($target_el.hasClass('columns')) target = "COLUMNS";
148         if ($target_el.hasClass('group')) target = "GROUP";
149         if ($target_el.hasClass('filter')) target = "FILTER";
150         
151        var index = $target_el.find('li.ui-draggable').index(
152                 $target_el.find('a[href="#' + dimension + '"]').parent() );
153         
154         var url = "/" + target + "/" + dimension + "/POSITION/" + index;
155         
156         this.workspace.query.action.del(url, {
157             success: this.workspace.query.run
158         });
159         
160         // Remove element
161         ui.draggable.addClass('deleted').remove();
162         
163         // Prevent workspace from getting this event
164         event.stopPropagation();
165         event.preventDefault();
166         return false;
167     },
168     
169     sort: function(event, ui) {
170 
171         var $target = $(event.target).parent().find('.dimension');
172             
173         var key = $target.attr('href').replace('#', '');
174         
175         var $li = $target.parent('.ui-draggable');
176         var index = $li.parent('.connectable').children().index($li);
177 
178         if ($li.parent('.connectable').parent().hasClass('columns')) target = "COLUMNS";
179         if ($li.parent('.connectable').parent().hasClass('group')) target = "GROUP";
180 
181  		$(event.target).cycleClass(["none","up","down"]);
182 
183  		var order = "NONE";	
184  		
185  		if($(event.target).hasClass("up")){
186  			order = "ASC";
187  		}
188  		if($(event.target).hasClass("down")){
189  			order = "DESC";
190  		}
191 
192  		this.workspace.query.action.post("/" + target + "/" + key + "/POSITION/" + index + "/SORT/" + order , { 
193             success: this.workspace.query.run
194         });
195     
196     },
197     
198     selections: function(event, ui) {
199         // Determine dimension
200         var $target = $(event.target).hasClass('sprite') ?
201             $(event.target).parent().find('.dimension') :
202             $(event.target);
203             
204         var key = $target.attr('href').replace('#', '');
205         
206         var $li = $target.parent('.ui-draggable');
207         var index = $li.parent('.connectable').children().index($li);
208  
209         if($target.parents('.fields_list').attr('title')=='COLUMNS')
210 		{
211 
212         // Launch column config dialog
213         (new ColumnConfigModal({
214             target: $target,
215             index: index,
216             name: $target.text(),
217             key: key,
218             workspace: this.workspace
219         })).open();
220 		}
221 
222         if($target.parents('.fields_list').attr('title')=='FILTERS')
223 		{        
224         // Launch selections dialog
225         (new SelectionsModal({
226             target: $target,
227             name: $target.text(),
228             key: key,
229             workspace: this.workspace
230         })).open();
231         }
232         
233         // Prevent default action
234         try {
235             event.preventDefault();
236         } catch (e) { }
237         return false;
238     }
239 });
