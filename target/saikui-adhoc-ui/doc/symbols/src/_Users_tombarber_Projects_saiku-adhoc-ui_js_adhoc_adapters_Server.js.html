  4 ;(function (window) {
  5 
  6   var
  7     characters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=',
  8     fromCharCode = String.fromCharCode,
  9     INVALID_CHARACTER_ERR = (function () {
 10       // fabricate a suitable error object
 11       try { document.createElement('$'); }
 12       catch (error) { return error; }}());
 13 
 14   // encoder
 15   window.Base64 || (
 16   window.Base64 = { encode: function (string) {
 17     var
 18       a, b, b1, b2, b3, b4, c, i = 0,
 19       len = string.length, max = Math.max, result = '';
 20 
 21     while (i < len) {
 22       a = string.charCodeAt(i++) || 0;
 23       b = string.charCodeAt(i++) || 0;
 24       c = string.charCodeAt(i++) || 0;
 25 
 26       if (max(a, b, c) > 0xFF) {
 27         throw INVALID_CHARACTER_ERR;
 28       }
 29 
 30       b1 = (a >> 2) & 0x3F;
 31       b2 = ((a & 0x3) << 4) | ((b >> 4) & 0xF);
 32       b3 = ((b & 0xF) << 2) | ((c >> 6) & 0x3);
 33       b4 = c & 0x3F;
 34 
 35       if (!b) {
 36         b3 = b4 = 64;
 37       } else if (!c) {
 38         b4 = 64;
 39       }
 40       result += characters.charAt(b1) + characters.charAt(b2) + characters.charAt(b3) + characters.charAt(b4);
 41     }
 42     return result;
 43   }});
 44

 47 /**
 48  * Model which handles AJAX calls to the Saiku Server
 49  * If you want to hook the UI up to something besides the Saiku Server,
 50  * this is the class which you want to override.
 51  * @returns {SaikuServer}
 52  */
 53 Backbone.sync = function(method, model, options) {
 54     var params;
 55     methodMap = {
 56         'create': "POST",
 57         'read': "GET",
 58         'update': "PUT",
 59         'delete': "DELETE"
 60     };
 61     
 62     // Generate AJAX action
 63     var type = methodMap[method];
 64     var url = Settings.REST_URL
 65         + (_.isFunction(model.url) ? model.url() : model.url);
 66 
 67     // Prepare for failure
 68     options.retries = 0;
 69     
 70     var failure = function(jqXHR, textStatus, errorThrown) {
 71         options.retries++;
 72         
 73         if (options.retries >= 10) {
 74             Application.ui.block("Could not reach server. Please try again later...");
 75             if (options.error) {
 76                 options.error(jqXHR, textStatus, errorThrown);
 77             }
 78         } else {
 79             var delay = Math.pow(options.retries, 2);
 80             Application.ui.block("Having trouble reaching server. Trying again in " + delay + " seconds...");
 81             setTimeout(function() {
 82                 $.ajax(params);
 83             }, delay * 1000);
 84         }
 85     };
 86     
 87     var success = function(data, textStatus, jqXHR) {
 88         if (options.retries > 0) {
 89             Saiku.ui.unblock();
 90         }
 91         
 92         options.success(data, textStatus, jqXHR);
 93     };
 94 
 95     // Default JSON-request options.
 96     
 97     params = {
 98       url:          url,
 99       type:         type,
100       cache:        false,
101       data:         model.attributes,
102       /*contentType:  'application/x-www-form-urlencoded',*/
103       contentType:  'application/json',  
104       dataType:     'json',
105       success:      success,
106       error:        failure,
107       beforeSend:   function(request) {
108         if (!Settings.PLUGIN) {
109           var auth = "Basic " + Base64.encode(
110               Saiku.session.username + ":" + Saiku.session.password
111           );
112           request.setRequestHeader('Authorization', auth);
113           }
114       }
115     };
116 
117 
118     // For older servers, emulate HTTP by mimicking the HTTP method with `_method`
119     // And an `X-HTTP-Method-Override` header.
120     if (Settings.BIPLUGIN || Backbone.emulateHTTP) {
121       if (type === 'PUT' || type === 'DELETE') {
122         if (Backbone.emulateHTTP) params.data._method = type;
123         params.type = 'POST';
124         params.beforeSend = function(xhr) {
125           xhr.setRequestHeader("X-HTTP-Method-Override", type);
126         };
127       }
128     }
129 
130     // Make the request
131     $.ajax(params);
132 };
