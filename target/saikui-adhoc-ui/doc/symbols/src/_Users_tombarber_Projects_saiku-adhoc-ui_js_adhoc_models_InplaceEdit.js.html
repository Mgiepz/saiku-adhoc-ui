1 var InplaceEdit = Backbone.Model.extend({
  2     initialize: function(args, options) {
  3 
  4         _.bindAll(this, "attach_listeners", "clicked_element");
  5 
  6         this.query = options.query;
  7 
  8         this.query.bind('report:render', this.attach_listeners);
  9     },
 10     
 11     attach_listeners: function(args) {  	
 12 		//TODO: be more specific, only childs of report_inner
 13         $(args.report.el).find(".saiku").click(this.clicked_element).hover(
 14         function() { $(this).addClass("report-hover"); },
 15         function() { $(this).removeClass("report-hover"); }
 16    		 );
 17         
 18         
 19         //simulate click on last edited element    
 20         if(this.query.lastEditElement!=null){    
 21         	$(args.report.el).find("." + this.query.lastEditElement).first().click();
 22         }
 23 
 24     },
 25     
 26     clicked_element: function(event) {
 27     	
 28     	$target = $(event.target);
 29     	
 30     	$target = $(event.target).is('span') ?
 31             $(event.target).parent() : $(event.target);
 32     	
 33     	var clazz = $target.attr('class').split(/\s+/);
 34     	
 35     	var elementClass;
 36     	
 37     	//find the relevant class
 38     	for (var i = 0; i < clazz.length; i++) {
 39                 var c = clazz[i];
 40                 if(c.substring(0, 3) == "rpt"){
 41                 	elementClass=c;
 42                 	break;
 43                 }
 44  		}
 45 
 46  		var splits = elementClass.split('-');
 47 
 48  		//all elements with the same class will be highlighted
 49  		if($('.' + elementClass).hasClass('adhoc-highlight')){
 50  			$('td').removeClass('adhoc-highlight');	
 51  			this.query.workspace.edit_panel.disable();
 52  		}else{
 53  			$('td').removeClass('adhoc-highlight');	
 54  			$('.' + elementClass).addClass('adhoc-highlight');	
 55  			this.query.workspace.trigger('report:edit', {
 56             	type: splits[1] , id: elementClass
 57         	});
 58  		}
 59  		
 60  		//If none is selected we dont want the selection to reappear after render
 61  		if(!$('.adhoc-highlight').length){
 62  			this.query.lastEditElement = null;
 63  		}		
 64     },
 65     
 66     check_input: function(event) {
 67         if (event.which == 13) {
 68             this.save_writeback(event);
 69         } else if (event.which == 27 || event.which == 9) {
 70             this.cancel_writeback(event);
 71         }
 72          
 73         return false;
 74     },
 75     
 76     save_writeback: function(event) {
 77         var $input = $(event.target).closest('input');
 78         this.set({
 79             value: $input.val(),
 80             position: $input.parent().attr('rel')
 81         });
 82         this.save();
 83         var value = $input.val();
 84         $input.parent().text(value);
 85     },
 86     
 87     cancel_writeback: function(event) {
 88         var $input = $(event.target).closest('input');
 89         $input.parent().text($input.parent().attr('alt'));
 90     },
 91     
 92     parse: function() {
 93         this.query.run();
 94     },
 95 	
 96     url: function() {
 97         return this.query.url() + "/edit/" + this.get('position') + 
 98             "/" + this.get('value'); 
 99     }
100 });
