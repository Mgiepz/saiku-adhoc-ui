1 /**
  2  * Dialog for member selections
  3  */
  4 var SelectionsModal = Modal.extend({
  5     type: "selections",
  6     
  7     buttons: [
  8         { text: "Save", method: "save" },
  9         { text: "Cancel", method: "close" }
 10     ],
 11     
 12     events: {
 13         'click a': 'call',
 14         'change #show_unique': 'show_unique'
 15     },
 16     
 17     initialize: function(args) {
 18         // Initialize properties
 19         _.extend(this, args);
 20         this.options.title = "Filter on " + this.name;
 21         this.message = "Fetching values...";
 22         this.show_unique = false;
 23         this.query = args.workspace.query;
 24         _.bindAll(this, "fetch_values", "populate", "finished");
 25         
 26         // Bind selection handlers
 27         _.extend(this.events, {
 28             'click div.selection_buttons a.form_button': 'move_selection'
 29         });
 30         
 31           
 32         // Resize when rendered
 33         this.bind('open', this.post_render);
 34         this.render();
 35         
 36         this.category = args.key.split('/')[1];
 37         this.column = args.key.split('/')[3];
 38         
 39         this.fetch_values();
 40 
 41     },
 42 
 43     fetch_values: function(category, column) {
 44 
 45         this.workspace.query.action.get("/FILTER/CATEGORY/" + this.category + "/COLUMN/" + this.column + "/result", { 
 46             success: this.populate
 47         });
 48     },
 49         
 50    populate: function(model, response) {
 51   
 52 		if(response.dataType=="Date"){
 53 			this.populate_datefilter(model, response);
 54 		}	    
 55 		else{
 56 			this.populate_multiselect(model, response);
 57 		}    
 58         // Show dialog
 59         Application.ui.unblock();
 60     },
 61     
 62     populate_datefilter: function(model, response) {
 63     	
 64      	var template = _.template($("#template-datefilter").html())(this);
 65      	$(this.el).find('.dialog_body').html(template);
 66 
 67         $(this.el).find('.datepicker_from').datepicker({
 68         	dateFormat: "dd.mm.yy",
 69         	changeMonth: true,
 70         	changeYear: true,
 71         	//minDate: startDate,
 72         	//maxDate: endDate,
 73         	onSelect: function(date, input){     
 74         		//alert(date);    
 75         	}
 76       	});
 77       	
 78       	$(this.el).find('.datepicker_to').datepicker({        	
 79         	dateFormat: "dd.mm.yy",
 80         	changeMonth: true,
 81         	changeYear: true,
 82         	//minDate: startDate,
 83         	//maxDate: endDate,
 84         	onSelect: function(date, input){    
 85         		//alert(date); 
 86         	}
 87       	});
 88      	
 89     },
 90  
 91     populate_multiselect: function(model, response) {
 92     	
 93    	try {
 94     		
 95     		var template = _.template($("#template-selections").html())(this); 
 96     		$(this.el).find('.dialog_body').html(template);
 97     		
 98 			this.available_values = JSON.parse(response.availableValues).resultset;
 99 			this.selected_values = response.selectedValues;
100 
101     		var used_values = [];
102             // Populate both boxes
103  
104  			if(this.selected_values!=null){
105             $(this.el).find('.used_selections select').removeAttr('disabled');
106             for (var j = 0; j < this.selected_values.length; j++) {
107                 var value = this.selected_values[j];
108                 //if (value.levelUniqueName == this.value.level &&
109                 //    member.type == "MEMBER") {
110                     $("<option />").text(value)
111                         .val(value)
112                         .appendTo($(this.el).find('.used_selections select'));
113                     used_values.push(value);
114                 //}
115             }
116  			} 
117             
118             // Filter out used values
119             this.available_values = _.select(this.available_values, function(obj) {
120                 return used_values.indexOf(obj[0]) === -1;
121             });
122            
123             $(this.el).find('.available_selections select').removeAttr('disabled');
124             for (var i = 0; i < this.available_values.length; i++) {
125 
126                 var value = this.available_values[i];
127 			
128 			//More elegant to remove null with _underscore?
129 			if(value[0]!=null){
130 				
131                 $("<option />").text(value[0])
132                     .val(value[0])
133                     .appendTo($(this.el).find('.available_selections select'));
134             }
135             
136             }
137         } catch (e) {
138             $(this.el).html("Could not load selections");
139         }
140     
141     },
142     
143     post_render: function(args) {
144         $(args.modal.el).parents('.ui-dialog').css({ width: "500px" });
145     },
146     
147     move_selection: function(event) {
148         var action = $(event.target).attr('id');
149         var $to = action.indexOf('add') !== -1 ? 
150             $(this.el).find('.used_selections select') :
151             $(this.el).find('.available_selections select');
152         var $from = action.indexOf('add') !== -1 ? 
153             $(this.el).find('.available_selections select') :
154             $(this.el).find('.used_selections select');
155         var $els = action.indexOf('all') !== -1 ? 
156             $from.find('option') :$from.find('option:selected');
157         $els.detach().appendTo($to);
158     },
159     
160     show_unique: function() {
161         $.each($(this.el).find('option'), function(i, option) {
162             var text = $(option).text();
163             $(option).text($(option).val());
164             $(option).val(text);
165         });
166         this.show_unique = ! this.show_unique;
167     },
168     
169     save: function() {
170         // Notify user that updates are in progress
171         var $loading = $("<div>Saving...</div>");
172         $(this.el).find('.dialog_body').children().hide();
173         $(this.el).find('.dialog_body').prepend($loading);
174    
175         var values = [];
176 
177 		//Date
178 		if($(this.el).find('.selection_buttons').length == 0){
179 			var val_from = $(this.el).find('.datepicker_from').val();
180 				values.push({
181 					value : val_from
182 				});
183 			var val_to = $(this.el).find('.datepicker_to').val();
184 				values.push({
185 					value : val_to
186 				});
187 			
188 		}
189 		//StringArray
190 		else{
191         	// Loop through selections
192 			$(this.el).find('.used_selections option')
193 				.each( function(i, selection) {
194 					var val = $(selection).text();
195 					values.push({
196 						value : val
197 					});
198 				});
199 		}	
200      
201         // Notify server
202         this.query.action.post('/FILTER/CATEGORY/' + this.category + '/COLUMN/' + this.column + '/VALUES', { 
203             success: this.finished,
204             data: values// JSON.stringify(values)
205         });
206         
207         return false;
208     },
209     
210     finished: function() {
211         $(this.el).dialog('destroy').remove();
212         this.query.run();
213     }
214 });
